library = []   
issued_books = set()   
def add_book():
    book_id = int(input("Enter Book ID: "))
    title = input("Enter Book Title: ")
    book = {"id": book_id, "title": title}  
    library.append(book)
    print("Book added successfully!\n")
def display_books():
    if not library:
        print("No books available!\n")
        return
    for book in library: 
        status = "Issued" if book["id"] in issued_books else "Available"
        print("ID:", book["id"], "| Title:", book["title"], "| Status:", status)
def issue_book():
    book_id = int(input("Enter Book ID to issue: "))
    
    for book in library:
        if book["id"] == book_id:  
            if book_id in issued_books:
                print("Book already issued!\n")
            else:
                issued_books.add(book_id)
                print("Book issued successfully!\n")
            return
    print("Book not found!\n")
def return_book():
    book_id = int(input("Enter Book ID to return: "))
    if book_id in issued_books:
        issued_books.remove(book_id)
        print("Book returned successfully!\n")
    else:
        print("Book was not issued!\n")
def show_books_tuple():
    print("Books as Tuples:")
    for book in library:
        print((book["id"], book["title"])) 
def count_books(n):
    if n == 0:
        return 0
    return 1 + count_books(n - 1)
while True:
    print("\n===== LIBRARY MENU =====")
    print("1. Add Book")
    print("2. Display Books")
    print("3. Issue Book")
    print("4. Return Book")
    print("5. Show Books (Tuple)")
    print("6. Count Books (Recursion)")
    print("7. Exit")
    
    choice = int(input("Enter choice: "))
    
    if choice == 1:
        add_book()
    elif choice == 2:
        display_books()
    elif choice == 3:
        issue_book()
    elif choice == 4:
        return_book()
    elif choice == 5:
        show_books_tuple()
    elif choice == 6:
        print("Total Books:", count_books(len(library)))
    elif choice == 7:
        print("Exiting program...")
        break
    else:
        print("Invalid choice!")
